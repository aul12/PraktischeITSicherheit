\section{What is Intel SGX}

\begin{frame}
    \frametitle{What is Intel SGX}
    \begin{itemize}
        \item \visible<1->{Software Guard Extension (SGX for short)}
        \item \visible<2->{Protects secrets in ``enclaves`` i.e. isolated execution environments}
        \item \visible<3>{Security policies enforced in CPU $\Rightarrow$ even OS or BIOS cannot access enclaves}
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Features of Intel SGX}
    \begin{itemize}
        \item \visible<1->{Secure computations within enclaves}
        \begin{itemize}
            \item \visible<2->{Backed by encrypted memory pages (Enclave Page Cache, EPC)}
            \item \visible<3->{Only addressable by the corresponding enclave}
        \end{itemize}
        \item \visible<4->{Computations run directly in CPU $\Rightarrow$ no dedicated TPM needed \newline $\Rightarrow$ fast}
        \item \visible<5->{Local and remote attestation}
        \item \visible<6>{Data sealing}
    \end{itemize}
    

\end{frame}
